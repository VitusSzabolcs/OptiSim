34double ThickLens::computeF(double n, double d, double r_left, double r_right){
35 if(r_left == 0.0 || r_right == 0.0) {
36 throw OptiSimError("ERROR: \tThe radius of the surface cannot be 0.");
37 }
38
39 double term1 = isinf(r_left) ? 0.0 : 1.0 / r_left;
40 double term2 = isinf(r_right) ? 0.0 : 1.0 / r_right;
41
42 double term3 = 0.0;
43 if (!isinf(r_left) && !isinf(r_right)) {
44 term3 = ((n - 1.0) * d) / (n * r_left * r_right);
45 }
46
47 double finv = (n - 1.0) * (term1 - term2 + term3);
48
49 if (abs(finv) < numeric_limits<double>::epsilon()) {
50 return numeric_limits<double>::infinity();
51 } else {
52 return 1.0 / finv;
53 }
54}

183Image ThickLens::Calculate(ImagingSubject is){
184 double H_left = computeHLeft();
185 double H_right = computeHRight();
186
187 double d_is;
188 if (isinf(is.getX())) {
189 d_is = numeric_limits<double>::infinity();
190 } else {
191 d_is = H_left - is.getX();
192 }
193
194 double y_is = is.getY();
195
196 double d_im;
197 double y_im;
198 bool is_real;
199
200 if (isinf(d_is)) {
201 d_im = f;
202 y_im = 0.0;
203 is_real = (f > 0);
204 } else {
205 double denominator = d_is - f;
206
207 if (abs(denominator) < numeric_limits<double>::epsilon()) {
208 y_im = numeric_limits<double>::infinity();
209
210 if (f > 0) {
211 d_im = numeric_limits<double>::infinity();
212 is_real = true;
213 } else {
214 d_im = -numeric_limits<double>::infinity();
215 is_real = false;
216 }
217 } else {
218 d_im = (f * d_is) / denominator;
219 y_im = -d_im / d_is * y_is;
220 is_real = d_im > 0;
221 }
222 }
223
224 return Image(H_right + d_im, y_im, is_real);
225}
