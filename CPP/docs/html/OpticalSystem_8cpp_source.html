180 ray_coord["ray_1"].y.push_back(LS->getY());

194 imageSequence.push_back(img);
195 }
196 // rays intersect at final image
197 ray_coord["ray_1"].x.push_back(img.getX());
198 ray_coord["ray_1"].y.push_back(img.getY());

334void OpticalSystem::NextRayCoords(OpticalObject* ActualLens, Image ActualImage, string which){
335 double x_image = ActualImage.getX();
336 double y_image = ActualImage.getY();
337 double x_ray = ray_coord[which].x.back();
338 double y_ray = ray_coord[which].y.back();
339 double x_lens = ActualLens->getX();
340
341 double a = (y_image-y_ray)/(x_image-x_ray);
342 double b = y_ray-a*x_ray;
343 double x_ray_new = x_lens;
344 double y_ray_new = a*x_ray_new+b;
345
346 ray_coord[which].x.push_back(x_ray_new);
347 ray_coord[which].y.push_back(y_ray_new);
348}
