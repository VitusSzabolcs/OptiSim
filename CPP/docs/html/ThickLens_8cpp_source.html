25double ThickLens::computeF(double n, double d, double r_left, double r_right){
26 if(r_left == 0.0 || r_right == 0.0) {
27 throw OptiSimError("ERROR: \tThe radius of the surface cannot be 0.");
28 }
29
30 double term1 = isinf(r_left) ? 0.0 : 1.0 / r_left;
31 double term2 = isinf(r_right) ? 0.0 : 1.0 / r_right;
32
33 double term3 = 0.0;
34 if (!isinf(r_left) && !isinf(r_right)) {
35 term3 = ((n - 1.0) * d) / (n * r_left * r_right);
36 }
37
38 double finv = (n - 1.0) * (term1 - term2 + term3);
39
40 if (abs(finv) < numeric_limits<double>::epsilon()) {
41 return numeric_limits<double>::infinity();
42 } else {
43 return 1.0 / finv;
44 }
45}

175 double H_left = computeHLeft();
176 double H_right = computeHRight();
177
178 double d_is;
179 if (isinf(is.getX())) {
180 d_is = numeric_limits<double>::infinity();
181 } else {
182 d_is = H_left - is.getX();
183 }
184
185 double y_is = is.getY();
186
187 double d_im;
188 double y_im;
189 bool is_real;
190
191 if (isinf(d_is)) {
192 d_im = f;
193 y_im = 0.0;
194 is_real = (f > 0);
195 } else {
196 double denominator = d_is - f;
197
198 if (abs(denominator) < numeric_limits<double>::epsilon()) {
199 y_im = numeric_limits<double>::infinity();
200
201 if (f > 0) {
202 d_im = numeric_limits<double>::infinity();
203 is_real = true;
204 } else {
205 d_im = -numeric_limits<double>::infinity();
206 is_real = false;
207 }
208 } else {
209 d_im = (f * d_is) / denominator;
210 y_im = -d_im / d_is * y_is;
211 is_real = d_im > 0;
212 }
